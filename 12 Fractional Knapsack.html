        void GreedyKnapsack(int num, int m)
        {
            float  p[num], w[num], ratio[num];
            int objects[num], i, j, k, temp;
            printf("\nEnter profit(p) & weight(w):\n");
            for(i=1; i<=num; i++)
            {
                scanf("%f %f", &p[i], &w[i]);
            }
            for(i=1; i<=num; i++)
            {
                ratio[i]=p[i]/w[i];
            }
            printf("\nObjects Profit  Weight\n");
            for(i=1; i<=num; i++)
            {
                printf("%d\t%.2f\t%.2f\n",i,p[i],w[i]);
            }
            for(i=1; i<=num; i++)
            {
                for(j=i+1; j<=num; j++)
                {
                    if(ratio[i]<ratio[j])
                    {
                        temp=ratio[j];
                        ratio[j]=ratio[i];
                        ratio[i]=temp;
        
                        temp = p[j];
                        p[j]=p[i];
                        p[i]=temp;
        
                        temp=w[j];
                        w[j]=w[i];
                        w[i]=temp;
                    }
                }
            }
            int u=m;
            float x[num], tp=0.0;
            for(i=1; i<=num; i++)
            {
                x[i]=0.0;
            }
            for(i=1; i<=num; i++)
            {
                if(w[i]>u)
                {
                    break;
                }
                else
                {
                    x[i]=1.00;
                    u = u - w[i];
                }
            }
            if(i<=num)
            {
                x[i]=u/w[i];
            }
            for(i=1; i<=num; i++)
            {
                tp = tp+(p[i]*x[i]);
            }
            printf("\nSolution vector(xi): ");
            for(i=1; i<=num; i++)
            {
                printf("%.2f ",x[i]);
            }
            printf("\n\nMaximum profit: %.2f\n",tp);
        }
