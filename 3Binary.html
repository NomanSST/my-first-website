        #include<stdio.h>

            int binary_search(int a[],int x, int left, int right)
            {
                int i,mid,count=0;
                for(i=left;i<=right;i++)
                {
                    mid=(left+right)/2;
                    if(a[mid]==x)
                        return mid;
                    else if(a[mid]<x)
                        left = mid+1;
                    else{
                        right=mid-1;
                }
                count++;
            
                }
                    return -1;
                printf("%d", count);
            }
            
            int main()
            {
                int i,n,x;
                int a[]={1,2,3,4,5};
                n=sizeof(a)/sizeof(a[0]);
                x=4;
                int r=binary_search(a,x,0,n-1);
                if(r==-1)
                    printf("not found");
                else
                    printf("Found");
