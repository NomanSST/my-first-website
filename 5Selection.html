        #include<stdio.h>

            void selection_sort(int a[], int n)
            {
                int i,j,index_min, temp;
                for(i=0;i<n-1;i++)
                {
                    index_min = i;
                    for(j=i+1;j<n;j++)
                    {
                        if(a[j]<a[index_min])
                        {
                            index_min=j;
                        }
                    }
                    if(index_min !=i)
                    {
                        temp=a[index_min];
                        a[index_min]=a[i];
                        a[i]=temp;
                    }
                }
            }
            
            int main()
            {
                int a[]={19, 8, 11,-34,-6,-11, 3, 16, 14, 6, 12, 2};
                int n=sizeof(a)/sizeof(a[0]);
                selection_sort(a,n);
                printf("A: ");
                for(int i=0;i<n;i++)
                {
                    printf("%d ",a[i]);
                }
            }
