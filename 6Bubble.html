So, here is the code:

        #include<stdio.h>
            void bubble_sort(int a[], int n)
            {
                int i,j,temp;
                for(i=0;i<n-1;i++)
                {
                    for(j=0;j<n-i-1;j++)
                    {
                        if(a[j]>a[j+1])
                        {
                            temp = a[j];
                            a[j]=a[j+1];
                            a[j+1]=temp;
                        }
                    }
                }
            }

            int main()
            {
                int i,n,a[]={5, 9, 2, -5,6,0,15, 1};
                n = sizeof(a)/sizeof(a[0]);
                bubble_sort(a,n);
                printf("A[]: ");
                for(i=0;i<n;i++)
                    printf("%d ", a[i]);
            }
