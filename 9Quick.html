    void quick_sort(int arr[], int low, int high)
    {
        if(low<high)
        {
            int p=partition(arr, low, high);
            quick_sort(arr, low, p-1);
            quick_sort(arr, p+1, high);
        }
    }
    int partition(int arr[], int low, int high)
    {
        int i,j, pivot,temp,temp1;
        pivot=arr[high];
        i=low-1;
        for(j=low; j<high; j++)
        {
            if(arr[j]<=pivot)
            {
                i=i+1;
                temp=arr[j];
                arr[j]=arr[i];
                arr[i]=temp;
            }
        }
        temp=arr[i+1];
        arr[i+1]=arr[high];
        arr[high]=temp;
    
        return (i+1);
    }
