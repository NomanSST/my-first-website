0/1 Knapsack(Dynamic Programming): Dynamic knapsack

So, here is the code:

    #include<stdio.h>

        int max(int a, int b)
        {
            if(a>b)
                return a;
            else
                return b;
        }
        
        int KS(int W, int pft[], int wt[], int n)
        {
            int i,j;
            int K[n+1][W+1];
            for(i=0; i<=n; i++)
            {
                for(j=0; j<=W; j++)
                {
                    if(i==0||j==0)
                        K[i][j]=0;
                    else if(wt[i-1]<=j)
                        K[i][j]=max(pft[i-1]+ K[i-1][j-wt[i-1]], K[i-1][j]);
                    else
                        K[i][j]=K[i-1][j];
                }
            }
            return K[n][W];
        }
        
        
        int main()
        {
            int i, n, W;
            printf("How many items: ");
            scanf("%d", &n);
            int pft[20], wt[20];
            printf("Enter profit and weight\n");
            for(i=0; i<n; i++)
            {
                printf("item-%d: ", i++);
                scanf("%d%d", &pft[i], &wt[i]);
            }
            printf("Enter the size of the Knapsack: ");
            scanf("%d", &W);
            printf("\nThe optimal profit: %d",KS(W, pft, wt, n));
        }
